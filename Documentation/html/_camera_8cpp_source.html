231 void Camera::UpdateViewMatrix()
232 {
233  if(mViewDirty)

235  XMVECTOR R = XMLoadFloat3(&mRight);
236  XMVECTOR U = XMLoadFloat3(&mUp);
237  XMVECTOR L = XMLoadFloat3(&mLook);
238  XMVECTOR P = XMLoadFloat3(&mPosition);
239 
240  // Keep camera's axes orthogonal to each other and of unit length.
241  L = XMVector3Normalize(L);
242  U = XMVector3Normalize(XMVector3Cross(L, R));
243 
244  // U, L already ortho-normal, so no need to normalize cross product.
245  R = XMVector3Cross(U, L);
246 
247  // Fill in the view matrix entries.
248  float x = -XMVectorGetX(XMVector3Dot(P, R));
249  float y = -XMVectorGetX(XMVector3Dot(P, U));
250  float z = -XMVectorGetX(XMVector3Dot(P, L));
251 
252  XMStoreFloat3(&mRight, R);
253  XMStoreFloat3(&mUp, U);
254  XMStoreFloat3(&mLook, L);
255
