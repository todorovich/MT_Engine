102  static UINT CalcConstantBufferByteSize(UINT byteSize)
103  {
104  // Constant buffers must be a multiple of the minimum hardware
105  // allocation size (usually 256 bytes). So round up to nearest
106  // multiple of 256. We do this by adding 255 and then masking off
107  // the lower 2 bytes which store all bits < 256.
108  // Example: Suppose byteSize = 300.
109  // (300 + 255) & ~255
110  // 555 & ~255
111  // 0x022B & ~0x00ff
112  // 0x022B & 0xff00
113  // 0x0200
114  // 512
115  return (byteSize + 255) & ~255;
116  }
