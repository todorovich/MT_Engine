285 void DirectXRenderer::create_root_signature()
286 {
287  // Shader programs typically require resources as input (constant buffers,
288  // textures, samplers). The root signature defines the resources the shader
289  // programs expect. If we think of the shader programs as a function, and
290  // the input resources as function parameters, then the root signature can be
291  // thought of as defining the function signature.
292 
293  // Root parameter can be a table, root descriptor or root constants.
294  CD3DX12_ROOT_PARAMETER slotRootParameter[1];
295 
296  // Create a single descriptor table of CBVs.
297  CD3DX12_DESCRIPTOR_RANGE cbvTable;
298  cbvTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_CBV, 1, 0);
299  slotRootParameter[0].InitAsDescriptorTable(1, &cbvTable);
300 
301  // A root signature is an array of root parameters.
302  CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(1, slotRootParameter, 0, nullptr,
303  D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
304 
305  // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
306  ComPtr<ID3DBlob> serializedRootSig = nullptr;
307  ComPtr<ID3DBlob> errorBlob = nullptr;
308  HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
309  serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
310 
311  if (errorBlob != nullptr)
312  {
313  ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
314  }
315 
316  ThrowIfFailed(hr);
317 
318  ThrowIfFailed(dx_device->CreateRootSignature(
319  0,
320  serializedRootSig->GetBufferPointer(),
321  serializedRootSig->GetBufferSize(),
322  IID_PPV_ARGS(&dx_root_signature)));
323 }

440 void DirectXRenderer::flush_command_queue()
441 {
442  // Advance the fence value to mark commands up to this fence point.
443  current_fence_index++;
444 
445  // Add an instruction to the command queue to set a new fence point. Because we
446  // are on the GPU timeline, the new fence point won't be set until the GPU finishes
447  // processing all the commands prior to this Signal().
448  ThrowIfFailed(dx_command_queue->Signal(fence.Get(), current_fence_index));
449 
450  // Wait until the GPU has completed commands up to this fence point.
451  if (fence->GetCompletedValue() < current_fence_index)
452  {
453  HANDLE eventHandle = CreateEventEx(nullptr, false, false, EVENT_ALL_ACCESS);
454 
455  // Fire event when GPU hits current fence.
456  ThrowIfFailed(fence->SetEventOnCompletion(current_fence_index, eventHandle));
457 
458  // Wait until the GPU hits current fence event is fired.
459  WaitForSingleObject(eventHandle, INFINITE);
460 
461  CloseHandle(eventHandle);
462  }
463 }

558 void DirectXRenderer::resize(int client_width, int client_height)
559 {
560  if (client_width != _client_width || client_height != _client_height)
561  {
562  _client_width = client_width;
563  _client_height = client_height;
564 
565  assert(dx_device);
566  assert(dx_swap_chain);
567  assert(dx_command_list_allocator);
568 
569  // Flush before changing any resources.
570  flush_command_queue();
571 
572  ThrowIfFailed(dx_command_list->Reset(dx_command_list_allocator.Get(), nullptr));
573 
574  // Release the previous resources we will be recreating.
575  for (int i = 0; i < _swap_chain_buffer_count; ++i)
576  swap_chain_buffer[i].Reset();
577  depth_stencil_buffer.Reset();
578 
579  // Resize the swap chain.
580  ThrowIfFailed(dx_swap_chain->ResizeBuffers(
581  _swap_chain_buffer_count,
582  _client_width, _client_height,
583  mBackBufferFormat,
584  DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH));
585 
586  current_back_buffer = 0;
587 
588  CD3DX12_CPU_DESCRIPTOR_HANDLE rtvHeapHandle(dx_rtv_heap->GetCPUDescriptorHandleForHeapStart());
589  for (UINT i = 0; i < _swap_chain_buffer_count; i++)
590  {
591  ThrowIfFailed(dx_swap_chain->GetBuffer(i, IID_PPV_ARGS(&swap_chain_buffer[i])));
592  dx_device->CreateRenderTargetView(swap_chain_buffer[i].Get(), nullptr, rtvHeapHandle);
593  rtvHeapHandle.Offset(1, rtv_descriptor_size);
594  }
595 
596  // Create the depth/stencil buffer and view.
597  D3D12_RESOURCE_DESC depthStencilDesc;
598  depthStencilDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
599  depthStencilDesc.Alignment = 0;
600  depthStencilDesc.Width = _client_width;
601  depthStencilDesc.Height = _client_height;
602  depthStencilDesc.DepthOrArraySize = 1;
603  depthStencilDesc.MipLevels = 1;
604  depthStencilDesc.Format = mDepthStencilFormat;
605  depthStencilDesc.SampleDesc.Count = _4x_msaa_state ? 4 : 1;
606  depthStencilDesc.SampleDesc.Quality = _4x_msaa_state ? (_4x_msaa_quality - 1) : 0;
607  depthStencilDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
608  depthStencilDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL;
609 
610  D3D12_CLEAR_VALUE optClear;
611  optClear.Format = mDepthStencilFormat;
612  optClear.DepthStencil.Depth = 1.0f;
613  optClear.DepthStencil.Stencil = 0;
614  ThrowIfFailed(dx_device->CreateCommittedResource(
615  &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
616  D3D12_HEAP_FLAG_NONE,
617  &depthStencilDesc,
618  D3D12_RESOURCE_STATE_COMMON,
619  &optClear,
620  IID_PPV_ARGS(depth_stencil_buffer.GetAddressOf())));
621 
622  // Create descriptor to mip level 0 of entire resource using the format of the resource.
623  dx_device->CreateDepthStencilView(depth_stencil_buffer.Get(), nullptr, get_depth_stencil_view());
624 
625  // Transition the resource from its initial state to be used as a depth buffer.
626  dx_command_list->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(depth_stencil_buffer.Get(),
627  D3D12_RESOURCE_STATE_COMMON, D3D12_RESOURCE_STATE_DEPTH_WRITE));
628 
629  // Execute the resize commands.
630  ThrowIfFailed(dx_command_list->Close());
631  ID3D12CommandList* cmdsLists[] = { dx_command_list.Get() };
632  dx_command_queue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
633 
634  // Wait until resize is complete.
635  flush_command_queue();
636 
637  // Update the viewport transform to cover the client area.
638  mScreenViewport.TopLeftX = 0;
639  mScreenViewport.TopLeftY = 0;
640  mScreenViewport.Width = static_cast<float>(_client_width);
641  mScreenViewport.Height = static_cast<float>(_client_height);
642  mScreenViewport.MinDepth = 0.0f;
643  mScreenViewport.MaxDepth = 1.0f;
644 
645  mScissorRect = { 0, 0, _client_width, _client_height };
646 
647  _client_aspect_ratio = static_cast<float>(_client_width) / _client_height;
648 
649  // The window resized, so update the aspect ratio and recompute the projection matrix.
650  engine::get_current_camera().SetLens(0.25f * MathHelper::Pi, engine::get_aspect_ratio(), 1.0f, 1000.0f);
651  //XMStoreFloat4x4(&mProj, P);
652  }
